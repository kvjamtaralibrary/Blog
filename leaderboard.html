  "rules": {
    "pending_comments": {
      ".write": "auth != null",

    },
    "approved_comments": {
      ".read": true,
      ".write": "auth.token.email == 'kvjamtaralibrary@gmail.com'"
    },
    "books": {
      ".read": true,
      ".write": "auth.token.email == 'kvjamtaralibrary@gmail.com'"
    },
    "leaderboard": {
      ".read": true,
      ".write": "auth != null"
    },
    "book_requests": {
      ".write": "auth != null",
      ".read": "auth.token.email == 'kvjamtaralibrary@gmail.com'"
    }
  }
